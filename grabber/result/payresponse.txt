    function getPoint(str) {
        str = Math.round(str * 100) / 100;
        if (str.toString() == 'NaN' || str == undefined)
            return 0.00;
        var idx = str.toString().indexOf(".");
        if (idx == -1) {
            str = str.toString() + ".00";
        }
        else if (idx > -1 && str.toString().length - idx == 2) {
            str = str.toString() + "0";
        }
        return parseFloat(str);
    }
